9. Notación BNF del Lenguaje Analizado

La notación BNF permite representar formalmente la gramática del subconjunto de JavaScript que reconoce el analizador sintáctico desarrollado. 


<programa> ::= <sentencia> <programa> | ε

<sentencia> ::= <declaracion>
              | <asignacion>
              | <impresion>
              | <condicional>
              | <bucle_for>
              | <bucle_while>

<declaracion> ::= <tipo_variable> IDENTIFICADOR "=" <valor> ";"

<tipo_variable> ::= "let" | "var" | "const"

<asignacion> ::= IDENTIFICADOR <operador_asignacion> <valor> ";"

<operador_asignacion> ::= "=" | "+=" | "-=" | "*=" | "/="

<impresion> ::= "console" "." "log" "(" <lista_valores> ")" ";"

<lista_valores> ::= <valor> <mas_valores>

<mas_valores> ::= "," <valor> <mas_valores> | ε

<condicional> ::= "if" "(" <condicion> ")" "{" <programa> "}" <opcional_else>

<opcional_else> ::= "else" "{" <programa> "}"
                   | "else" <condicional>
                   | ε

<bucle_for> ::= "for" "(" IDENTIFICADOR "=" <valor> ";" <condicion> ";" IDENTIFICADOR "++" ")" "{" <programa> "}"

<bucle_while> ::= "while" "(" <condicion> ")" "{" <programa> "}"

<condicion> ::= IDENTIFICADOR <operador_relacional> <valor>

<operador_relacional> ::= "==" | "!=" | "<" | ">" | "<=" | ">="

<valor> ::= IDENTIFICADOR
           | LITERAL_NUMERICO
           | LITERAL_DECIMAL
           | LITERAL_CADENA
           | LITERAL_BOOLEANO
