7. Gramática Incontextual (Gramática Libre de Contexto)

La siguiente gramática describe las estructuras sintácticas reconocidas por el analizador sintáctico desarrollado para JavaScript. 

<programa> ::= <sentencia> <programa>
              | ε

<sentencia> ::= <declaracion>
              | <asignacion>
              | <impresion>
              | <if>
              | <for>
              | <while>

<declaracion> ::= let IDENTIFICADOR = <valor> ;
                | var IDENTIFICADOR = <valor> ;
                | const IDENTIFICADOR = <valor> ;

<asignacion> ::= IDENTIFICADOR = <valor> ;

<impresion> ::= console . log ( <valor> ) ;

<if> ::= if ( <condicion> ) { <programa> }

<for> ::= for ( <asignacion> <condicion> ; IDENTIFICADOR ++ ) { <programa> }

<while> ::= while ( <condicion> ) { <programa> }

<condicion> ::= IDENTIFICADOR <operador_relacional> <valor>

<valor> ::= IDENTIFICADOR
           | LITERAL_NUMERICO
           | LITERAL_DECIMAL
           | LITERAL_CADENA
           | LITERAL_BOOLEANO

<operador_relacional> ::= ==
                         | !=
                         | <
                         | >
                         | <=
                         | >=
